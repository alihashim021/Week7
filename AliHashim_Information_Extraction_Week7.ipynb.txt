        "import re\n",
        "from datetime import datetime\n",
        "\n",
        "def extract_dates(text):\n",
        "    \"\"\"Extract dates in various formats (MM/DD/YYYY, DD-MM-YYYY, Month DD, YYYY).\"\"\"\n",
        "    patterns = [\n",
        "        r'\\d{1,2}/\\d{1,2}/\\d{4}',\n",
        "        r'\\d{1,2}-\\d{1,2}-\\d{4}',\n",
        "        r'\\b(?:Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec)[a-z]* \\d{1,2},? \\d{4}',\n",
        "        r'\\d{4}-\\d{2}-\\d{2}'\n",
        "    ]\n",
        "    dates = []\n",
        "    for pattern in patterns:\n",
        "        matches = re.findall(pattern, text)\n",
        "        dates.extend(matches)\n",
        "    return dates\n",
        "\n",

        "def extract_invoice_number(text):\n",
        "    \"\"\"Identify invoice or order numbers using common prefix patterns.\"\"\"\n",
        "    patterns = [r'INV-\\d{4}-\\d{3}', r'#\\d{5,}', r'ORDER-[A-Z0-9]+', r'Invoice (?:Number|#):?\\s*([A-Z0-9-]+)']\n",
        "    for pattern in patterns:\n",
        "        match = re.search(pattern, text, re.IGNORECASE)\n",
        "        if match:\n",
        "            return match.group(1) if match.groups() else match.group(0)\n",
        "    return None\n",

        "def extract_entities(text):\n",
        "    \"\"\"Categorize text into specific entities like Persons, Organizations, and Locations.\"\"\"\n",
        "    doc = nlp(text)\n",
        "    entities = {'persons': [], 'organizations': [], 'locations': [], 'dates': [], 'money': []}\n",
        "\n",
        "    for ent in doc.ents:\n",
        "        if ent.label_ == 'PERSON': entities['persons'].append(ent.text)\n",
        "        elif ent.label_ == 'ORG': entities['organizations'].append(ent.text)\n",
        "        elif ent.label_ in ['GPE', 'LOC']: entities['locations'].append(ent.text)\n",
        "        elif ent.label_ == 'DATE': entities['dates'].append(ent.text)\n",
        "        elif ent.label_ == 'MONEY': entities['money'].append(ent.text)\n",
